#初始化
 git init 
#添加到暂存区
 git add *
#查看工作区状态
 git status
#删除文件
 git rm 文件名
#查看版本
 git log
#查看版本变化
 git reflog
#提交到本地仓库
 git commit -m ''
#设置推送远程仓库路径
 git remote add origin https://github.com/taurus-wsk/wks-git-test.git
#将本地主分支设置为默认推送仓库
 git push -u origin master
#将文件推送到远程仓库
 git push  
#将文件从远程仓库拉取
 git pull  
#将全部文件从远程仓库下载
 git clone
#版本回退
 git reset --hard id
#撤销修改
 git restore 文件名
#查看不同
 git diff
#从暂存区撤销
 git restore --staged
#创建并且切换到分支
 git checkout -b dev
#切换回主分支
 git checkout master
#合并分支
 git merge dev
#删除分支
 git branch -D dev
#分支合并图
 git log --graph
#暂存保存
git stash
#查看暂存保存列表
git stash list
#将保存的数据恢复
git stash apply
#删除远程 origin/A 分支 
git push origin --delete A
#创建远程分支并提交,也可从origin/master切换回origin/wsk-dev
git push --set-upstream origin wsk-dev
#删除已经deleted的文件
git rm $(git ls-files --deleted)
#在分支上设置追踪，提交到master
